# Arithmetic-operator-demo
Learning project in C to understand how arithmetic operator works?

#include <stdio.h>

void main() {

    int a=20, b=12;
    
    printf("addition=%d",a+b);
    
    printf("substraction=%d",a-b);
    
    printf("multiplication=%d",a*b);
    
    printf("division=%d",a/b);
    
    printf("module=%d",a/b);

}
